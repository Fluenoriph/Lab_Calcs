CREATE TABLE protocol (
    id INTEGER AUTOINCREMENT,
   	number TEXT UNIQUE NOT NULL CHECK (number != ''),
	date_of_research TEXT NULL,
	protocol_date TEXT NULL,
	work_type TEXT NULL,
	object_name TEXT NOT NULL CHECK (name != ''),
    object_address TEXT NOT NULL CHECK (address != ''),
	employee TEXT NULL,
	PRIMARY KEY (id, number)
	);








CREATE TABLE microclimate (
	id INTEGER PRIMARY KEY AUTOINCREMENT,
	ok_standart INTEGER NULL,
	no_standart INTEGER NULL,
	FOREIGN KEY (id) REFERENCES protocols(number) ON UPDATE CASCADE
	);
